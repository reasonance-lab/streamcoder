import streamlit as st

# Function to mix two colors
def mix_colors(color1, color2):
    mix = {
        ('red', 'yellow'): 'orange',
        ('yellow', 'red'): 'orange',
        ('red', 'blue'): 'purple',
        ('blue', 'red'): 'purple',
        ('yellow', 'blue'): 'green',
        ('blue', 'yellow'): 'green'
    }
    return mix.get((color1, color2), "Mix another combination!")

# Adding a title
st.title("Color Mixing Fun!")

# Dropdown selectors for primary colors
st.header("Pick two colors to mix")
primary_colors = ['red', 'yellow', 'blue']
color1 = st.selectbox('Pick the first color:', primary_colors)
color2 = st.selectbox('Pick the second color:', primary_colors)

# Button to mix colors
if st.button('Mix the colors!'):
    result = mix_colors(color1, color2)
    st.write(f'When you mix {color1} and {color2}, you get {result}!')

# Show common combinations
st.header("Common Color Mixes")
st.write("Red + Yellow = Orange")
st.write("Red + Blue = Purple")
st.write("Yellow + Blue = Green")
