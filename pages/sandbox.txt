import streamlit as st
import pandas as pd
import numpy as np

# Set page title
st.title("Fictitious Area Chart Example")

# Generate sample data
dates = pd.date_range(start="2023-01-01", end="2023-12-31", freq="D")
data = {
    "Category A": np.random.randint(100, 1000, size=len(dates)),
    "Category B": np.random.randint(50, 800, size=len(dates)),
    "Category C": np.random.randint(200, 1200, size=len(dates))
}

# Create DataFrame
df = pd.DataFrame(data, index=dates)

# Display area chart
st.area_chart(df)

# Add description
st.write("This area chart shows fictitious data for three categories over the year 2023.")

# Display raw data
if st.checkbox("Show raw data"):
    st.write(df)
