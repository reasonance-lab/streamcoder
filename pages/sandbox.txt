import requests
import json

def get_drug_info(drug_name):
    base_url = "https://api.fda.gov/drug/label.json"
    params = {
        "search": f"openfda.brand_name:{drug_name}",
        "limit": 1
    }

    response = requests.get(base_url, params=params)
    
    if response.status_code == 200:
        data = json.loads(response.text)
        if data['results']:
            return data['results'][0]
        else:
            return None
    else:
        return None

def format_drug_info(drug_info):
    formatted_info = ""
    
    if drug_info:
        # Add brand name
        if 'openfda' in drug_info and 'brand_name' in drug_info['openfda']:
            formatted_info += f"Brand Name: {drug_info['openfda']['brand_name'][0]}\n\n"
        
        # Add generic name
        if 'openfda' in drug_info and 'generic_name' in drug_info['openfda']:
            formatted_info += f"Generic Name: {drug_info['openfda']['generic_name'][0]}\n\n"
        
        # Add description
        if 'description' in drug_info:
            formatted_info += f"Description: {drug_info['description'][0]}\n\n"
        
        # Add indications and usage
        if 'indications_and_usage' in drug_info:
            formatted_info += f"Indications and Usage: {drug_info['indications_and_usage'][0]}\n\n"
        
        # Add warnings
        if 'warnings' in drug_info:
            formatted_info += f"Warnings: {drug_info['warnings'][0]}\n\n"
        
        # Add adverse reactions
        if 'adverse_reactions' in drug_info:
            formatted_info += f"Adverse Reactions: {drug_info['adverse_reactions'][0]}\n\n"
    else:
        formatted_info = "No information found for the given drug name."
    
    return formatted_info

def main():
    drug_name = input("Enter a drug name: ")
    drug_info = get_drug_info(drug_name)
    formatted_info = format_drug_info(drug_info)
    print(formatted_info)

if __name__ == "__main__":
    main()
